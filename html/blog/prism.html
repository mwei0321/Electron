function index(){
		        $count = $this->electron->getElectronList();
	            if($count > 0){
	                $page = new Page($count, 50);
	                $list = $this->electron->getElectronList([],$page->firstRow.','.$page->listRows);
	            }

	            $this->assign('list',$list);
	            $this->display();
		    }
